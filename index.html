// Sample combinedArray for demonstration const combinedArray = [ ['Red', 'red'], ['Green', 'green'], ['Blue', 'blue'], // Add more colors as needed ]; // Function to shuffle the array function shuffleArray(array) { for (let i = array.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [array[i], array[j]] = [array[j], array[i]]; } return array; } // Assuming 'boxes' is a NodeList of elements const boxes = document.querySelectorAll('.box'); // Adjust the selector as needed // Function to set initial text function setInitialText() { const shuffledArray = shuffleArray([...combinedArray]); boxes.forEach((box, index) => { box.innerHTML = box.style.backgroundColor; // Set initial inner text to the background color }); } // Function to refresh all boxes function refreshAllBoxes() { const shuffledArray = shuffleArray([...combinedArray]); boxes.forEach((box, index) => { box.innerHTML = box.style.backgroundColor; // Update box content to background color }); } // Example usage document.addEventListener('DOMContentLoaded', () => { setInitialText(); // Initial setup // You can call refreshAllBoxes() as needed });
